import csv
import collections
import itertools
import operator

def try_float(s):
    try:
        return float(s)
    except:
        return s

with open('results.csv', 'r') as f:
    reader = csv.DictReader(f)

    experiments = {}
    for experiment, results in itertools.groupby(reader, operator.itemgetter('Input.experiment')):
        experiments[experiment] = list(results)

%matplotlib inline
from matplotlib import pyplot as pl

for experiment_group in experiments.itervalues():
    settings = experiment_group[0]
    costs = {}
    for cost in 'rose', 'daisy', 'tulip', 'flower':
        costs[cost] = float(settings['Input.{0}_cost'.format(cost)])
    for cost, value in sorted(costs.items(), key=operator.itemgetter(1)):
        print '{0}:\t{1:.2f}'.format(cost.capitalize(), float(settings['Input.{0}_cost'.format(cost)])),
        print '!' if cost == 'flower' else ''
    
    f, axes = pl.subplots(1, 3, )
    f.set_size_inches(18, 6)
    
    for item, axis in zip(('rose', 'daisy', 'tulip'), axes):
        answers = [
            float(x['Answer.{}'.format(item)]) 
            for x in experiment_group 
            if float(x['Answer.sanity']) >= 0.75 # sanity check applied here
        ]
        axis.hist(answers, range=(0., 1.))
        axis.set_title("'Flower' ({[Input.flower_cost]}) instead of '{}' ({})".format(
                settings, item, settings['Input.{}_cost'.format(item)])
        )
    print '({} valid responses / {})'.format(len(answers), len(experiment_group))
    pl.show()
    print
    print
